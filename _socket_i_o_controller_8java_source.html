96 @EventListener(eventID = "onSocketIOConnect")
97 public void onClientConnect(SocketIOEventData e) {
98 log.info("New user connected. Session ID: " + e.getClient().getSessionId() + " - IP: " + e.getClient().getRemoteAddress());
99
100 SocketIOConnectEventData eventData = (SocketIOConnectEventData) e.getData();
101 DecodedJWT jwt = UtilFunctions.validateKeycloakToken(e.getClient().getHandshakeData().getHttpHeaders().get("token"), eventData.getKeycloakIssuerUrl());
102 PlayerEntity entity = playerService.readByKeycloakUID(jwt.getId());
103
104 if (entity == null) {
105 entity = new PlayerEntity();
106 entity.setKeycloakUID(jwt.getId());
107 entity.setDisplayName(jwt.getClaim("preferred_username").asString());
108
109 entity = playerService.create(entity);
110 }
111
112 clientMap.put(entity, e.getClient());
113 }
114
124 @EventListener(eventID = "onSocketIODisconnect")
125 public void onClientDisconnect(SocketIOEventData data) {
126 log.info("User disconnected. Session ID: " + data.getClient().getSessionId() + " - IP: " + data.getClient().getRemoteAddress());
127
128 clientMap.remove(clientMap.entrySet().stream().filter(player -> player.getValue().getSessionId() == data.getClient().getSessionId()).findFirst().get().getKey());
129 }
