41 public synchronized void registerListeners(Object... listeners) {
42 for (Object listener : listeners) {
43 for (Method method : listener.getClass().getDeclaredMethods()) {
44 if (!method.isAnnotationPresent(EventListener.class))
45 continue;
46
47 EventListener eventAnnotation = method.getAnnotation(EventListener.class);
48
49 if (!eventAnnotation.lobbySpecific()) {
50 eventListeners.putIfAbsent(eventAnnotation.eventID(), new HashMap<>());
51 eventListeners.get(eventAnnotation.eventID()).put(listener, method);
52 } else {
53 String lobbyCode = listener instanceof Game ? ((Game) listener).getLobby().getId() : ((Lobby) listener).getId();
54
55 gameListeners.putIfAbsent(lobbyCode, new HashMap<>());
56 gameListeners.get(lobbyCode).putIfAbsent(eventAnnotation.eventID(), new HashMap<>());
57 gameListeners.get(lobbyCode).get(eventAnnotation.eventID()).put(listener, method);
58 }
59 }

123 private synchronized Object fireGameEvent(String eventID, SocketIOEventData socketIOEventData) {
124 String lobbyCode = socketIOEventData.getClientRoomId();
125
126 if (lobbyCode == null)

129 if (!gameListeners.containsKey(lobbyCode))

131

150 private synchronized Object fireGlobalEvent(String eventID, SocketIOEventData socketIOEventData) {
151 if (!eventListeners.containsKey(eventID))
152 return null;
153
154 // We are aware that this creates a race condition where only one event returns anything and the other
155 // events are not executed at all. But after a careful evaluation of the issue, we have decided to leave
156 // it as it is and only fix it if we run into a problem.
157 for (Map.Entry<Object, Method> entry : eventListeners.get(eventID).entrySet())
158 return _fireEvent(entry.getKey(), entry.getValue(), socketIOEventData);
159
160 return null;
161 }

173 private synchronized Object _fireEvent(Object object, Method method, SocketIOEventData socketIOEventData) {
