128 public Map<PlayerEntity, List<WhiteCardEntity>> dealWhiteCards() {
129 for (Map.Entry<PlayerEntity, List<WhiteCardEntity>> hand : playerHands.entrySet()) {
130 while (hand.getValue().size() < MAX_HAND_SIZE) {
131 hand.getValue().add(freeWhiteCards.remove(freeWhiteCards.size() - 1));
132 if (freeWhiteCards.isEmpty()) {
133 freeWhiteCards = new ArrayList<>(cardSet.getWhiteCards());
134 freeWhiteCards.removeAll(getAllCardsInHands());
135 }
136 SocketIOController.getInstance().getClientByPlayer(hand.getKey()).sendEvent("onDealWhiteCards", hand.getValue());
137 }
138 }
